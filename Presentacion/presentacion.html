02 · Comparación
Sin librería · implementación manualheappop O(n)
class heapq:

    def heappush(frontera, elemento):
        frontera.append(elemento)

    def heappop(frontera):
        menor_indice = 0

        for i in range(len(frontera)):
            if frontera[i] < frontera[menor_indice]:
                menor_indice = i

        return frontera.pop(menor_indice)

def uniform_cost_search(graph, start, goal):
    node = start
    costo_inicial = 0

    frontera = []
    heapq.heappush(frontera, (costo_inicial, node))

    explorados = set()

    parent = {}
    parent[start] = None

    costos = {}
    costos[start] = 0

    if start == goal:
        return parent, 0

Reconstrucción completa del camino

Se recorren los padres desde la meta hasta el inicio y luego se invierte la lista.

06 · reconstruct_path
reconstruct_path(sol, start, goal)función completa
def reconstruct_path(sol, start, goal):
    if start == goal:
        return [start]

    if goal not in sol:
        return []

    path = [goal]
    s = goal

    while s:
        padre = sol.get(s)

        if padre is None:
            break

        path.append(padre)
        s = padre

        if s == start:
            break

    return list(reversed(path))
